node实现一个简易git-commit

Authors
  • Name
    Later

前置知识

执行git commit时会生成git对象中的另外两个对象--tree (树) 对象和commit (提交) 对象，都保存在.git/objects目录中 ，和存储 (数据) 对象类似，文件名都是一串hash值

我们新建一个目录，初始化git git init, 创建一个目录+文件和一个文件

mkdir dir 
echo "hello,world" > dir/file
echo "hello,world" > test

然后执行git add . && git commit -m 1

存储 (数据) 对象

git ls-files -s查看index中的内容, 可以看到针对没个文件都有一个存储对象

100644 2d832d9044c698081e59c322d5a2a459da546469 0       dir/file
100644 2d832d9044c698081e59c322d5a2a459da546469 0       test

git cat-file -p 2d832d9044c698081e59c322d5a2a459da546469可以看到存储对象的内容，就是我们创建文件的内容

commit (提交) 对象

接下来看提交对象, cat .git/refs/heads/master可以看到提交对象的文件名，然后git cat-file -p d34b4b70b9ffd02ca9f9e0793bca1ebf625bbb56可以看到提交对象的内容：

tree 7ba1b0ff40e0a1be5034fac5234feec65ab8d57f
author houbinbin <houbb@neocrm.com> 1628302598 +0800
committer houbinbin <houbb@neocrm.com> 1628302598 +0800


1

里边存着顶层树对象的信息 作者/提交者信息、前时间戳、一个空行，以及提交注释信息

git cat-file -p d34b4b70b9ffd02ca9f9e0793bca1ebf625bbb56 : commit

tree (树) 对象

通过 git cat-file -p 7ba1b0ff40e0a1be5034fac5234feec65ab8d57f可以看到树对象的内容

040000 tree 6674ad3a0df49acc4cc923b586bb1a8ddc232185    dir
100644 blob 2d832d9044c698081e59c322d5a2a459da546469    test

git cat-file -t 7ba1b0ff40e0a1be5034fac5234feec65ab8d57f: tree

源码简单实现

执行git commit时，需要从index文件中取出内容递归写树对象，树对象写完后记录顶层的树对象内容到commit对象中，写commit对象，commit对象写完后找到.git/HEAD文件中的分支名称，在.git/refs/heads/master文件中更新刚写完的commit对象hash(commitID)

所有异常先不考虑，实现核心逻辑

const fs = require('fs');
const path = require('path');
const { createHash } = require('crypto');
const { set, isString } = require('lodash');


const hash = (str) => createHash('sha1').update(str).digest('hex');


const gitDir = path.resolve(process.cwd(), '.ngit');
// 从index中取出所有数据 { 'test/file': '7317399078361350034a7b769c6ea479ca1ebc4a', 'test2/file1' '1323131 }
  const indexObj = fs
    .readFileSync(path.resolve(gitDir, 'index'), 'utf8')
    .split('\n')
    .filter((i) => i)
    .reduce((acc, cur) => {
      acc[cur.split(' ')[2]] = cur.split(' ')[0];
      return acc;
    }, {});
// 处理index数据 { test: { file: '7317399078361350034a7b769c6ea479ca1ebc4a' } }
const treeObject = Object.keys(indexObj).reduce((acc, cur) => {
  const objectPath = cur.split('/').join('.');
  set(acc, objectPath, indexObj[cur]);
  return acc;
}, {});


  // 获取树 tree hash
const genTreeHash = (tree, isWrite) => {
  const treeStr =
    Object.keys(tree)
      .map((key) =>
        isString(tree[key]) ? `blob ${tree[key]} ${key}` : `tree ${genTreeHash(tree[key], isWrite)} ${key}`,
      )
      .join('\n') + '\n';
  const treeHash = hash(treeStr);
  if (isWrite) {
    fs.writeFileSync(path.resolve(gitDir, 'objects', treeHash), treeStr);
  }
  return treeHash;
};


// 获取顶层的树对象，并写入树对象
const treeHash = genTreeHash(treeObject, true);
// 获取当前分支名
const heads = fs.readFileSync(path.resolve(gitDir, 'HEAD'), 'utf8').match('refs/heads/(.+)')[1];
let commitParentHash = '';
if (fs.existsSync(path.resolve(gitDir, 'refs/heads', heads))) {
  commitParentHash = fs.readFileSync(path.resolve(gitDir, 'refs/heads', heads), 'utf8');
}

const commitStr = `tree ${treeHash}\n${
  commitParentHash ? 'parent ' + commitParentHash + '\n' : ''
}Date ${new Date().toString()}\n\n${message}\n`;
const commitHash = hash(commitStr);
// 写commit 对象
fs.writeFileSync(path.resolve(gitDir, 'objects', commitHash), commitStr);


// ref里写值
fs.writeFileSync(path.resolve(gitDir, 'refs/heads', heads), commitHash);


return `[${heads} ${commitHash}] ${message}`;
