node实现一个简易git-rm

Authors
  • Name
    Later
    Twitter

前置知识

git rm可以理解为rm+git add,这里需要处理一个异常，就是暂存区的文件已经更新就不能删除

源码简单实现

实现思路：

要删除的文件如果没有在暂存区，直接删除，如果已经在暂存区了，分两种情况：

  1. 刚初始化的仓库，还没有commit，那暂存区的文件肯定是已经更新过的了，不能删除
  2. 已经有过commit过的仓库，先找到当前分支的commitID，根据commitID里的顶层树对象找到所有文件信息，对比最后一次提交里的文件hash和暂存区的文件hash，如果一致，直接删除，不一致，文件已经变更，不能删除

如果要删除的文件不存在，直接返回

const fs = require('fs');
const path = require('path');
const { isString } = require('lodash')
const gitDir = path.resolve(process.cwd(), '.ngit');


// 获取文件相对路径
const relativeFilePath = path.relative(process.cwd(), path.resolve(process.cwd(), filePath));
// 文件不存在，直接返回
if (!fs.existsSync(path.resolve(process.cwd(), filePath))) {
  return `pathspec ${filePath} did not match any files`
}
// index 数据转 object  index 中的 object { 'test/file': '7317399078361350034a7b769c6ea479ca1ebc4a' }
const idx = fs
  .readFileSync(path.resolve(gitDir, 'index'), 'utf8')
  .split('\n')
  .filter((i) => i)
  .reduce((acc, cur) => {
    acc[cur.split(' ')[2]] = cur.split(' ')[0];
    return acc;
  }, {});
// index 中匹配到的文件
const matchedFile = Object.keys(idx).find((p) => p === relativeFilePath);


if (!matchedFile) {
  // 删除文件
  fs.unlinkSync(path.resolve(process.cwd(), filePath));
  return `rm ${relativeFilePath}`;
}
// 对比index 中的hash对象 和 heads中的 hash 对象，找到已经变更的文件列表
// 先找到当前分支
const currentBranch = fs.readFileSync(path.resolve(gitDir, 'HEAD'), 'utf8').match('ref: (refs/heads/.+)')[1];
// 1. 分支指向存在 已经提交过，暂存区文件hash 和 上次提交树hash 对比   2. 分支指向不存在 直接返回不能删除
if (fs.existsSync(path.resolve(gitDir, currentBranch))) {
  // 获取当前分支的commitId
  const currentCommitId = fs.readFileSync(path.resolve(gitDir, currentBranch), 'utf8');
  // 根据当前分支的commitId 反推出当前 树对象 {a: {b: 1}}
  const getTreeObjectByTreeHash = (treeHash) => {
    const treeObject = {};
    const objectPath = path.resolve(gitDir, 'objects', treeHash);
    fs.readFileSync(objectPath, 'utf8').split('\n').filter(i => i).forEach((line) => {
      const lineTokens = line.split(/ /);
      treeObject[lineTokens[2]] = lineTokens[0] === 'tree' ? getTreeObjectByTreeHash(lineTokens[1]) : lineTokens[1];
    });
    return treeObject;
  };
  const treeHash = fs.readFileSync(path.resolve(gitDir, 'objects', currentCommitId), 'utf8').split('\n')[0].split('tree ')[1];
  const currentTreeObj = getTreeObjectByTreeHash(treeHash);
  // {a: {b: 1}} => {'a/b': 1}
  const flattenTree = (tree, obj = {}, prefix = '') => {
    Object.keys(tree).forEach((dir) => {
      const filePath = path.join(prefix, dir);
      if (isString(tree[dir])) {
        obj[filePath] = tree[dir];
      } else {
        flattenTree(tree[dir], obj, filePath);
      }
    });
    return obj;
  };
  // refs/master 中的 object { 'test/file': '7317399078361350034a7b769c6ea479ca1ebc4a' }
  const trees = flattenTree(currentTreeObj);
  // 判断hash值是否变化
  if (trees[relativeFilePath] !== idx[relativeFilePath]) {
    return `the following file has changes staged in the index => ${relativeFilePath}`;
  }


  // 删除文件
  fs.unlinkSync(path.resolve(process.cwd(), filePath));
  // 更新索引
  delete idx[relativeFilePath];
  // 写索引
  const indexStr =
    Object.keys(idx)
      .map((k) => `${idx[k]} 0 ${k.split(',')[0]}`)
      .join('\n') + '\n';
  fs.writeFileSync(path.resolve(gitDir, 'index'), indexStr);
  return `rm ${relativeFilePath}`;
} else {
  return `the following file has changes staged in the index => ${relativeFilePath}`;
}
